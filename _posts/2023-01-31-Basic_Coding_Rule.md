int binarySearch(int data[], int n, int target)
{
    int start = 0;
    int end = n;
    int mid;
    while (end >= start)
    {
        mid = (end + start) / 2;
        if (data[mid] == target)
            return 1;
        else if (data[mid] > target)
            end = mid - 1;
        else
            start = mid + 1;
    }
    return 0;
}
